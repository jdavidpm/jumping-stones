unsigned long long jumps(int E)
{
   int i;

   if (E == N) return 1;
   if (E > N) return 0;
   if (DP[E] != -1) return DP[E];

   DP[E] = 0;
   for (i = 1; i <= k; ++i)
      DP[E] += jumps(E + i);

   return DP[E];
}
